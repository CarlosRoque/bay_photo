def self.define(*args, start_at: 0)
  fail "Must have at least one symbol in args list" if args.nil? or args.empty?
  fail "All args must be Symbols" unless args.all? { |arg| arg.is_a? Symbol }

  Class.new(EnumInstance) do
    @args = args
    @start_at = start_at
  end
end
